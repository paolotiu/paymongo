```ts
export type ErrorSubCode =
  | 'card_expired'
  | 'cvc_invalid'
  | 'generic_decline'
  | 'fraudulent'
  | 'insufficient_funds'
  | 'processor_blocked'
  | 'lost_card'
  | 'stolen_card'
  | 'processor_unavailable'
  | 'blocked';
```
